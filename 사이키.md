void PortInit()
{
	DDRB = 0xff;
}


void Timer_8bit_Init(void)
{

	TCCR0 = (1<<CS02) | (1<<CS01) | (1<<CS00); //표준모드, 분주비 1024
	TCNT0 = 100; // 10msec
	TIMSK = (1<<TOIE0); //타이머 오버플로 인터럽트 허용
	sei();
}

void Timer_16bit_Init(void)
{

	TCCR1A = (1<<COM1A1) | (1<<COM1B1) | (1<<COM1C1);
	TCCR1A |= (1<<COM1A0) | (1<<COM1B0) | (1<<COM1C0);
	/// 채널 a,b,c 열어줌 //  OCnA/OCnB/OCnC -> high level
	//inverting mode

	TCCR1A |= (1<<WGM11) |  (1<<WGM12) |  (1<<WGM13);  
	//mode 14번 fast pwm, 열어준 채널에 pwm 발생

	TCCR1B = (1<<WGM13) | (1<<WGM12); //열어준 채널에 pwm 발생
	TCCR1B |= (1<<CS12) | (1<<CS10); //분주비 1024

	ICR1 = 255; // ICR1의 TOP 값 (약61hz)  TOP:ICRn
}
void colorLedCon()
{
	//1.빨강 255 시작 -> 녹색 증가
	//2.녹색 255 -> 빨강 감소
	//3.빨강 0되면 파랑 증가
	//4.파랑 255 -> 녹색 감소
	//5.녹색 0 -> 빨강 증가
	//6.빨강 255 -> 파랑감소
	static int red=255,green=0,blue=0,state=0;
	switch(state)
	{
		case 0:
			if(green<255) green++;
			else state=1;
		break;
		case 1:
			if(red>0) red--;
			else state=2;
		break;
		case 2:
			if(blue<255) blue++;
			else state=3;
		break;
		case 3:
			if(green>0) green--;
			else state=4;
		break;
		case 4:
			if(red<255) red++;
			else state=5;
		break;
		case 5:
			if(blue>0) blue--;
			else state=6;
		break;
		case 6:
			red=255;green=0;blue=0;state=0;
		break;
	}

	OCR3A = blue; //blue
 	OCR3B = green; ///green
    OCR3C = red; //red

}

void colorLedConRGB(int red,int green, int blue)
{
 OCR3A = blue; //blue
 OCR3B = green; ///green
 OCR3C = red; //red

}
